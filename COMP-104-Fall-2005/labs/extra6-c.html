// extra6-1.cpp
// Tic Tac Toe ("Two-end-snake" in Chinese)
// Computer strategy:
//      1) If the computer can win on the next move, do it,
//      2) If the user can win on the next move, block it
//      3) Pick a random unoccupied square.

#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int board[4][4];                             // the game board matrix: 0=unpicked, 1=computer, 5=user
void printBoard();
bool draw();
bool computerWin();
bool playerWin();
bool blockWin();
void pickRandom();

int main(){
    int r, c;

    srand(time(0));                     //initialize random number generator
    for(r=1;r<=3;r++)                   // initialize the board
        for(c=1;c<=3;c++)
            board[r][c]=0;
    while(true){
        // computer's turn:
        if(draw()){
            cout << "Draw!" << endl;
            return 0;
        }
        if(computerWin()){
            cout<<"Computer's move: " << endl;
            printBoard();
            cout << "Computer wins!" << endl;
            return 0;
        }
        else if(blockWin()){
            cout<<"Computer's move: " << endl;
            printBoard();
        }
        else{
            pickRandom();
            cout<<"Computer's move: " << endl;
            printBoard();
        }

        // player's turn:
        if(draw()){
            cout << "Draw!" << endl;
            return 0;
        }
        while(true){
            cout << "User's move (give row, a space, and column): ";
            cin >> r >> c;
            if(r>=1 && r<=3 && c>=1 && c<=3 && board[r][c]==0){
                board[r][c] = 5;
                break;
            }
            cout << "Illegal move! Re-enter." << endl;
        }
        printBoard();
        if(playerWin()){
            cout << "User wins!" << endl;
            return 0;
        }
    }
    return 0;
}

//------------------------------------------------------------------------------------------------------
// returns true if all taken, false otherwise
bool draw(){
        int r, c;
        for(r=1; r<=3; r++)
            for(c=1; c<=3; c++)
                if(board[r][c]==0)
                    return false;
        return true;                            // all must be taken
}

//------------------------------------------------------------------------------------------------------
// returns true if computer can win (and makes move), false otherwise
bool computerWin(){
        int r, c;
        for(r=1; r<=3; r++)                    // check row win
            if(board[r][1]+board[r][2]+board[r][3]==2){
                for(c=1; c<=3; c++)
                if(board[r][c]==0)
                    board[r][c] = 1;
                return true;
            }
        for(c=1; c<=3; c++)                    // check column win
            if(board[1][c]+board[2][c]+board[3][c]==2){
                for(r=1; r<=3; r++)
                if(board[r][c]==0)
                    board[r][c] = 1;
                    return true;
            }
        if(board[1][1]+board[2][2]+board[3][3]==2){   // check right diagonal
            for(r=1; r<=3; r++)
                if(board[r][r]==0)
                    board[r][r] = 1;
                return true;
        }
        if(board[1][3]+board[2][2]+board[3][1]==2){   // check left diagonal
            for(r=1; r<=3; r++)
                if(board[r][4-r]==0)
                    board[r][4-r] = 1;
                return true;
        }
        return false;
}

        for(c=1; c<=3; c++)                  // check column win

        if(board[1][1]+board[2][2]+board[3][3]==15)     // check right diagonal
            return true;
        if(board[1][3]+board[2][2]+board[3][1]==15)     // check left diagonal
            return true;

int map[4][4];                             // the array for holding the tic tac toe game
int computer_win=-1;                       // 0 -- computer win, 1 -- computer lose, 2-- Draw

        int com=computer;                  // 0 -- for using O ,   1 -- for using X

            for (j=1;j<=3;j++) {
                cout<<" ";
                if ((map[i][j]==1) && (com==0)) cout<<"O";
                else if ((map[i][j]==1) && (com==1)) cout<<"X";
                else if ((map[i][j]==5) && (com==0)) cout<<"X";
                else if ((map[i][j]==5) && (com==1)) cout<<"O";
                else cout<<" ";

                else
                    cout<<" | ";
           }
           if (i==3)
               cout<<endl;
           else

        }
}

// Check whether user can win the game, although it won't happen
// it can be used in the future, when the program is developed to
// user play with user

void check_win() {
        int i;
        for (i=1;i<=3;i++) {
            if (map[i][1]==5 && map[i][1]==map[i][2] &&
                map[i][2]==map[i][3]){
                computer_win=1;                 //computer loses on row
                return;
            }
        }
        for (i=1;i<=3;i++) {
            if (map[1][i]==5 && map[1][i]==map[2][i] &&
                map[2][i]==map[3][i]){
                computer_win=1;                 //computer loses on column
                return;
            }
        }
        if (map[1][1]==5 && map[1][1]==map[2][2] && map[2][2]==map[3][3]){
            computer_win=1;                     //computer loses on diagonal
            return;
        }
        if (map[1][3]==5 && map[1][3]==map[2][2] && map[2][2]==map[3][1]){
            computer_win=1;                     //computer loses on diagonal
            return;
        }
}

// check for is there any move which can lead to win for a line

int going_to_win(int x, int y, char direction) {
        int temp;
        switch (direction) {
            case 'v':                          // check for column
                temp=map[x][y]+map[x+1][y]+map[x+2][y];
                if (temp==2) {
                    if (map[x][y]==0)          // return co-ordinate
                        return x*10+y;
                    else if (map[x+1][y]==0)
                        return (x+1)*10+y;
                    else if (map[x+2][y]==0)
                        return (x+2)*10+y;
                    else return -1;
                }
                break;
            case 'h':                         // check for row
                temp=map[x][y]+map[x][y+1]+map[x][y+2];
                if (temp==2) {
                    if (map[x][y]==0)         // return co-ordinate
                        return x*10+y;
                    else if (map[x][y+1]==0)
                        return x*10+(y+1);
                    else if (map[x][y+2]==0)
                        return x*10+(y+2);
                    else return -1;
                }
                break;
            case 'd':                         // check for diagnoal
                temp=map[x][y]+map[2][2]+map[(x+2)%4][(y+2)%4];
                if (temp==2) {
                    if (map[x][y]==0)
                        return x*10+y;
                    else if (map[2][2]==0)
                        return 2*10+2;
                    else if (map[(x+2)%4][(y+2)%4]==0)
                        return ((x+2)%4)*10+((y+2)%4);
                    else return -1;
                }
                break;
            default:
                return -1;
        }
        return -1;
}

// checking the move which lead to win for whole map
int go_win() {

        int score=-1;
        int i;

        for (i=1;i<=3;i++) {
                score=going_to_win(i,1,'h');    // check for row
                if (score!=-1)
                        return score;
        }
        for (i=1;i<=3;i++) {
                score=going_to_win(1,i,'v'); // check for column
                if (score!=-1)
                        return score;
        }
        score=going_to_win(1,1,'d');
        if (score!=-1)
                return score;
        score=going_to_win(1,3,'d');
        if (score!=-1)
                return score;
        return -1;              // no winning move
}

// auto_move is a general methods to decide the next move
// 1. See whether any move lead to win
// 2. See whether any move lead to avoid lose
// 3. see whether center left
// 4. see whether corner left
// 5. see whether middle left

void auto_move() {
        int temp=-1;
        int x=-1,y=-1;  // for holding new move co-oridnate

        temp=go_win();  //check for winning step

        if (temp!=-1) {
                x=temp/10;      // extract out the x-axis cooridnate
                y=temp%10;      // extract out the y-axis cooridnate
                map[x][y]=1;    // fill the map
                computer_win=0; // computer win
                return;
        }

        temp=go_lose(); //check for losing step
        if (temp!=-1) {
                x=temp/10;      // extract out the x-axis cooridnate
                y=temp%10;      // extract out the y-axis cooridnate
                map[x][y]=1;    // fill the map
                return;
        }

        // fill center
        if (map[2][2]==0) {     // if the center didn't filled, fill it first.
                map[2][2]=1;
                return;
        }

        // fill corner
        if (map[1][1]==0) {     // if the corner (1,1) didn't filled, fill it.
                map[1][1]=1;
                return;
        }
        if (map[3][3]==0) {     // if the corner (3,3) didn't filled, fill it.
                map[3][3]=1;
                return;
        }
        if (map[1][3]==0) {     // if the corner (1,3) didn't filled, fill it.
                map[1][3]=1;
                return;
        }
        if (map[3][1]==0) {     // if the corner (3,1) didn't filled, fill it.
                map[3][1]=1;
                return;
        }

        // fill middle
        if (map[1][2]==0) {     // if the middle (1,2) didn't filled, fill it.
                map[1][2]=1;
                return;
        }
        if (map[2][3]==0) {     // if the middle (2,3) didn't filled, fill it.
                map[2][3]=1;
                return;
        }
        if (map[3][2]==0) {     // if the middle (3,2) didn't filled, fill it.
                map[3][2]=1;
                return;
        }
        if (map[3][3]==0) {     // if the middle (3,3) didn't filled, fill it.
                map[3][3]=1;
                return;
        }
        return;
}
