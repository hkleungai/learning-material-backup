       cout << "Simple Card Game\n";
       cout << "----------------\n";
       cout << "Computer cards: ";
       for (int i=0; i<3; i++)
       {
            ccards[i] = rand() % 13 + 1;
            if (ccards[i] == 1)
                 cout << "A" << " " ;
            else if((ccards[i] <= 10) && (ccards[i] != 1))
                 cout << ccards[i] << " " ;

                 cout << "J" << " " ;

                 cout << "Q" << " " ;
            else if(ccards[i] == 13)
                 cout << "K" << " " ;

       cout << endl;
       cout << "Computer score: ";
       for (i=0; i<3; i++)
       {
           if (ccards[i] > 10)
                ctotal += 10;
           else
                ctotal += ccards[i];
       }
       ctotal = ctotal%10;
       cout << ctotal << endl;

       cout << "User cards: ";
       for (i=0; i<3; i++)
       {

                 cout << "A" << " " ;
            else if((ucards[i] <= 10) && (ucards[i] != 1))
                 cout << ucards[i] << " " ;
            else if(ucards[i] == 11)
                 cout << "J" << " " ;
            else if(ucards[i] == 12)
                 cout << "Q" << " " ;
            else if(ucards[i] == 13)
                 cout << "K" << " " ;
       }
       cout << endl;
       cout << "User score: ";
       for (i=0; i<3; i++)
       {
           if (ucards[i] > 10)
                utotal += 10;
           else
                utotal += ucards[i];
       }
       utotal = utotal%10;
       cout << utotal << endl;

       if (ctotal == utotal)
            cout << "Draw Game!" << endl;

            cout << "Computer wins!" << endl;

            cout << "User wins!" << endl;

}

// Simple Card Game with repeat mode
#include <iostream>
#include <cstdlib>              // contains the random number generator rand()
#include <ctime>                // contains time() to seed the random generator
using namespace std;

int main(){
    int ccards[3], ucards[3];
    int ctotal = 0, utotal = 0;
    int cwins = 0, uwins = 0;
    char resp;
    srand(time(0));             // initialize random number generator
    cout << "Simple Card Game\n";
    cout << "----------------\n";
    do {
        cout << "Computer cards: ";
        for (int i=0; i<3; i++)
        {
            ccards[i] = rand() % 13 + 1;
            if (ccards[i] == 1)
                 cout << "A" << " " ;
            else if((ccards[i] <= 10) && (ccards[i] != 1))
                 cout << ccards[i] << " " ;
            else if(ccards[i] == 11)
                 cout << "J" << " " ;
            else if(ccards[i] == 12)
                 cout << "Q" << " " ;
            else if(ccards[i] == 13)
                 cout << "K" << " " ;
        }
        cout << endl;
        cout << "Computer score: ";
        for (i=0; i<3; i++)
        {
           if (ccards[i] > 10)
                ctotal += 10;
           else
                ctotal += ccards[i];
        }
        ctotal = ctotal%10;
        cout << ctotal << endl;

        cout << "User cards: ";
        for (i=0; i<3; i++)
        {
            ucards[i] = rand() % 13 + 1;
            if (ucards[i] == 1)
                 cout << "A" << " " ;
            else if((ucards[i] <= 10) && (ucards[i] != 1))
                 cout << ucards[i] << " " ;
            else if(ucards[i] == 11)
                 cout << "J" << " " ;
            else if(ucards[i] == 12)
                 cout << "Q" << " " ;
            else if(ucards[i] == 13)
                 cout << "K" << " " ;
        }
        cout << endl;
        cout << "User score: ";
        for (i=0; i<3; i++)
        {
           if (ucards[i] > 10)
                utotal += 10;
           else
                utotal += ucards[i];
        }
        utotal = utotal%10;
        cout << utotal << endl;

        if (ctotal == utotal)
            cout << "Draw Game!" << endl;
        else if (ctotal > utotal)
        {
            cout << "Computer wins!" << endl;
            cwins++;
        }
        else
        {
            cout << "User wins!" << endl;
            uwins++;
        }
        cout << "Another game? (Y/N): ";
           cin >> resp;
   } while ((resp == 'Y' || resp == 'y'));
   cout << endl;
   cout << "Total user wins: " << uwins << endl;
   cout << "Total computer wins: " << cwins << endl;
   return 0;
}


// extra4-3.cpp
// 15 coins
/*  Basic strategy:
        15 remaining: computer picks 2
        13 remaining: user picks 1, 2, or 3
        12 remaining: computer picks 3
        11 remaining: computer picks 2
        10 remaining: computer picks 1
         9 remaining: user picks 1, 2, or 3
         8 remaining: computer picks 3
         7 remaining: computer picks 2
         6 remaining: computer picks 1
         5 remaining: user picks 1, 2, or 3
         4 remaining: computer picks 3
         3 remaining: computer picks 2
         2 remaining: computer picks 1
         1 remaining: user picks 1 (user loses)

   Can generalize above strategy so that:
        number of coins computer picks = 4 - number of coins user picks
*/
#include <iostream>
using namespace std;
int main(){
    int coins = 15;
    int compcoins, usercoins=2;

    cout << "Starting with 15 coins" << endl;
    while(true){
        compcoins = 4 - usercoins;
        coins -= compcoins;
        cout << "Computer takes " << compcoins << ", " << coins << " coins remaining" << endl;
        if(coins==1){
            cout << "User is left with final coin, user loses" << endl;
            return 0;
        }
        cout << "How many coins do you want: ";
        cin >> usercoins;
        coins -= usercoins;
        cout << "User takes " << usercoins << ", " << coins << " coins remaining" << endl;
    }
}
