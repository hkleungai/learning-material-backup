//extra8-1.cpp
//File Average

#include <iostream>
#include <fstream>
using namespace std;

void main(){
    ifstream ins;
    double average, next;
    double sum = 0.0;
    int count = 0;                        // Count the number of input numbers
    ins.open("number.dat");
    while(ins >> next){
        sum += next;                    // Calculate the sum of the number
        count++;
    }
    ins.close();
    average = sum/count;
    cout << "The average of these " << count << " number(s): ";
    cout << average << endl;
}


//extra8-2.cpp
//Pig Latin Translator (by Chow Kwan Cheuk)

#include <iostream>
#include <string>
using namespace std;

bool vowel(char);

//**************************************************************************

int main() {
    string instr, outstr;
    char next;
    int space[200];
    int t_space = 0;
    int i, j, width;

    cout << "Enter sentence: ";
    while (cin.get(next)) {
        if (next == '\n')
            break;
        else
            instr += next;
    }
    instr += ' ';

    space[0] = -1;
    t_space++;
    for(i=0; i<instr.length(); i++)
        if (instr[i] == ' ') {
            space[t_space] = i;
            i++;
            t_space++;
        }
    for(i=0; i<t_space-1; i++) {
        if (vowel(instr[space[i]+1])) {
            outstr += (instr.substr(space[i]+1,space[i+1]-space[i]-1) + "-ay ");
        }
        else {
            width = 0;
            while(!vowel(instr[space[i]+1+width]))
                width++;
            if(isupper(instr[space[i]+1]))
                outstr += toupper(instr[space[i]+1+width]);
            else
                outstr += tolower(instr[space[i]+1+width]);
            outstr += instr.substr(space[i]+1+width+1,space[i+1]-space[i]-1-width-1) + '-';
            outstr += tolower(instr[space[i]+1]);
            for (j = 1; j < width; j++)
                outstr += instr[space[i]+1+j];
            outstr += "ay ";
        }
    }

    cout << "Translation: " << outstr << endl;
    return 0;
}

//**************************************************************************

bool vowel(char a) {
    if ( (a == 'a') || (a == 'e') || (a == 'i') || (a == 'o') || (a == 'u') ||
         (a == 'A') || (a == 'E') || (a == 'I') || (a == 'O') || (a == 'U') )
         return 1;
    else return 0;
}
