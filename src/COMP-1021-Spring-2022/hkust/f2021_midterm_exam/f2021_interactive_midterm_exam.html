sections = [100, 100, 100, 100, 100, 100, 100, 100, 100, 99, 98, 96]
nihk = [14, 2, 13, 3, 2, 17, 5, 4, SEE QUESTION, 5, 2, 6]

sectionTotal = sum(sections)

result = (nihkTotal / sectionTotal) * 100.0
