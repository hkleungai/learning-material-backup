    void reverse_queue(int_queue q)
{
	int_stack s;
	while (!q.empty())
	{
		int a = q.front();		
		q.dequeue();
		s.push(a);
	}
	while (!s.empty())
	{
		int a = s.top();	
		s.pop();
		q.enqueue(a);
	}
