Exceptions in ML

* An example of defining exceptions in function head:

exception Head; (* exception binding *)

fun head(nil) = raise Head
|   head(x::l) = x;


* An example of handling exceptions:

fun head2 l = head(l) handle Head => 0;


* An example of multiple exceptions:

exception Theirs and Mine;

fun f(x) = if x=0 then raise Mine else raise Theirs;

f(0) handle Mine => 7;

f(1) handle Mine => 7;

(f(1) handle Mine =>7) handle Theirs =>8;
